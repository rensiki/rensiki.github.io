## 내 자바스크립트 게임

<div id="game"></div>

<script>
  // 간단한 예시: 버튼 클릭 게임
  const gameDiv = document.getElementById('game');
  gameDiv.innerHTML = '<button id="btn">클릭!</button> <span id="score">0</span>';
  let score = 0;
  document.getElementById('btn').onclick = function() {
    score++;
    document.getElementById('score').textContent = score;
  };
</script>
